def search_company_by_inn(inn: str) -> Optional[Dict]:

    try:

    return None

def parse_okved_products(okved_codes: List[str]) -> Tuple[List[str], str]:
    """Анализирует ОКВЭД и возвращает список подходящих страховых продуктов"""
    all_products = set()
    reasons = []
    
    for code in okved_codes:
        # Берем первые 2 символа для категоризации
        category = code[:2]
        if category in OKVED_TO_INSURANCE:
            products = OKVED_TO_INSURANCE[category]["products"]
            reason = OKVED_TO_INSURANCE[category]["reason"]
            all_products.update(products)
            reasons.append(reason)
        else:
            # Пробуем по первой цифре
            first_digit = code[:1]
            if first_digit in OKVED_TO_INSURANCE:
                all_products.update(OKVED_TO_INSURANCE[first_digit]["products"])
    
    if not all_products:
        return DEFAULT_PRODUCTS, DEFAULT_REASON
    
    # Убираем дубликаты, сохраняя порядок
    unique_products = []
    for p in list(all_products):
        if p not in unique_products:
            unique_products.append(p)
    
    main_reason = reasons[0] if reasons else DEFAULT_REASON
    return unique_products[:5], main_reason

    # Информация о руководителе
    individuals = company.get("individuals", [])
    director = "Не указан"
    for person in individuals:
        if person.get("type") == "PrivatePerson":
            full_name = person.get("fullName", {})

            name = full_name.get("name", "")

def is_inn(text: str) -> bool:
    """Проверяет, является ли строка ИНН (10 или 12 цифр)"""
    return bool(re.match(r'^\d{10}$|^\d{12}$', text.strip()))

@bot.message_handler(func=lambda message: True)
def handle_company_query(message):
    query = message.text.strip()
    
    if not query:
        bot.reply_to(message, "❌ Введите ИНН или название компании")
        return
    
    # Отправляем уведомление о начале поиска
    status_msg = bot.reply_to(message, "🔍 Ищу информацию...")
    
    company = None
    
    if is_inn(query):
        # Поиск по ИНН
        company = search_company_by_inn(query)
    else:
        # Поиск по названию
        company = search_company_by_name(query)
    
    if company:
        report = format_company_report(company)
        bot.edit_message_text(
            report, 
            chat_id=status_msg.chat.id, 
            message_id=status_msg.message_id,
            parse_mode="Markdown"
        )
    else:
        # Если компания не найдена
        bot.edit_message_text(
            f"❌ Не удалось найти компанию по запросу: {query}\n\n"
            "Попробуйте:\n"
            "• Уточнить название\n"
            "• Использовать ИНН (10 или 12 цифр)\n"
            "• Написать часть названия",
            chat_id=status_msg.chat.id,
            message_id=status_msg.message_id
        )
